# Stuff To Do When You Move Prod to Stg (same formula for prod to dev)

# sync the db
drush @calacademy.prod sql-dump > /tmp/calacademy.prod.sql
drush @calacademy.stg sql-drop --yes
drush @calacademy.stg sql-cli < /tmp/calacademy.prod.sql --yes
drush @calacademy.stg cc all

# Varnish stuff
drush @calacademy.stg vset varnish_control_terminal 'www2-stg-cache:6082'

# More Varnish stuff
# The actual value is the read-protected contents of /etc/varnish/secret
drush @calacademy.stg vset varnish_control_key 'xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx'

# block googlebot
drush @calacademy.stg php-eval 'variable_set("robotstxt", "User-agent: *\nDisallow: /")'

# enable Stage File Proxy Module
drush @calacademy.stg en stage_file_proxy --yes
